1. Triangle Validator and Classifier

Problem Statement:

Write a program to accept the three sides of a triangle and display whether a valid triangle can be formed or not, if formed, whether it is an equilateral, isosceles or a scalene triangle and calculate its area.

Code:


import java.util.Scanner;
import java.lang.Math;

public class TriangleCalculator {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        System.out.println("Enter 3 sides:");
        double a = scanner.nextDouble();
        double b = scanner.nextDouble();
        double c = scanner.nextDouble();

        if ((a + b > c) && (a + c > b) && (b + c > a)) {
            System.out.println("Triangle is valid.");
            if (a == b && b == c) {
                System.out.println("It is an Equilateral triangle.");
            } else if (a == b || b == c || a == c) {
                System.out.println("It is an Isosceles triangle.");
            } else {
                System.out.println("It is a Scalene triangle.");
            }
            double s = (a + b + c) / 2;
            double area = Math.sqrt(s * (s - a) * (s - b) * (s - c));
            System.out.printf("Area is: %.2f\n", area);
        } else {
            System.out.println("Triangle is not valid.");
        }
        scanner.close();
    }
}

2. Fibonacci Series

Problem Statement:

Write a program using a for loop to print Fibonacci series of n terms where n is input by user (0 1 1 2 3 5 8 13 21....)

Code:


import java.util.Scanner;

public class FibonacciSeries {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        System.out.print("Enter number of terms: ");
        int n = scanner.nextInt();

        int t1 = 0, t2 = 1;
        System.out.print("Fibonacci Series: ");

        for (int i = 1; i <= n; ++i) {
            System.out.print(t1 + " ");
            int sum = t1 + t2;
            t1 = t2;
            t2 = sum;
        }
        System.out.println();
        scanner.close();
    }
}

3. Reverse an Integer

Problem Statement:

Write a program using a while loop that prompts the user to input an integer and then outputs the number with the digits reversed. (e.g., 12345 -> 54321)

Code:


import java.util.Scanner;

public class ReverseNumber {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        System.out.print("Enter an integer: ");
        int num = scanner.nextInt();
        
        int reversedNum = 0;

        while (num != 0) {
            int digit = num % 10;
            reversedNum = reversedNum * 10 + digit;
            num /= 10;
        }

        System.out.println("Reversed number: " + reversedNum);
        scanner.close();
    }
}

import java.util.Scanner;

class Rectangle {
    double length;
    double breadth;

    public void setDim(double len, double brd) {
        this.length = len;
        this.breadth = brd;
    }

        System.out.println("Area: " + rect.calculateArea());
        System.out.println("Perimeter: " + rect.calculatePerimeter());

    }
}

9. Bank Account (Menu Driven)

Problem Statement:

Design a menu-driven program for a bank account (Owner, AccNo, Balance, Type). Include a constructor and methods for deposit, withdraw, and display.

Code:


import java.util.Scanner;

class BankAccount {
    String ownerName;
    double balance;

    public BankAccount(String name, double initialBalance) {
        this.ownerName = name;
        this.balance = initialBalance;
    }

    public void deposit(double amount) {
        if (amount > 0) {
            this.balance += amount;
            System.out.println("Deposit successful. New balance: " + this.balance);
        }
    }

    public void withdraw(double amount) {
        if (amount > this.balance) {
            System.out.println("Insufficient funds.");
        } else {
            this.balance -= amount;
            System.out.println("Withdrawal successful. New balance: " + this.balance);
        }
    }

    public void displayDetails() {
        System.out.println("Owner: " + this.ownerName + ", Balance: " + this.balance);
    }
}

public class BankMenu {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        BankAccount account = new BankAccount("John Doe", 1000.00);
        int choice;

        do {
            System.out.println("\n--- Bank Menu ---");
            System.out.println("1. Deposit");
            System.out.println("2. Withdraw");
            System.out.println("3. Display Details");
            System.out.println("4. Exit");
            System.out.print("Enter choice: ");
            choice = scanner.nextInt();

            switch (choice) {
                case 1:
                    System.out.print("Enter deposit amount: ");
                    account.deposit(scanner.nextDouble());
                    break;
                case 2:
                    System.out.print("Enter withdraw amount: ");
                    account.withdraw(scanner.nextDouble());
                    break;
                case 3:
                    account.displayDetails();
                    break;
                case 4:
                    System.out.println("Exiting...");
                    break;
            }
        } while (choice != 4);
        scanner.close();
    }
}

12. Stack Operations

Problem Statement:

Write a program to implement stack operations (PUSH, POP, DISPLAY)

Code:


class Stack {
    private int[] stackArray;
    private int top;
    private int maxSize;

    public Stack(int size) {
        this.maxSize = size;
        this.stackArray = new int[maxSize];
        this.top = -1; // -1 means empty
    }

    public void push(int value) {
        if (top == maxSize - 1) {
            System.out.println("Stack Overflow!");
        } else {
            top++;
            stackArray[top] = value;
            System.out.println(value + " pushed.");
        }
    }

    public void pop() {
        if (top == -1) {
            System.out.println("Stack Underflow!");
        } else {
            int value = stackArray[top];
            top--;
            System.out.println(value + " popped.");
        }
    }

    public void display() {
        if (top == -1) {
            System.out.println("Stack is empty.");
        } else {
            System.out.print("Stack (top to bottom): ");
            for (int i = top; i >= 0; i--) {
                System.out.print(stackArray[i] + " ");
            }
            System.out.println();
        }
    }
}

public class StackDemo {
    public static void main(String[] args) {
        Stack myStack = new Stack(5);
        myStack.push(10);
        myStack.push(20);
        myStack.display();
        myStack.pop();
        myStack.display();
    }
}

13. Matrix Transpose

Problem Statement:

Write a program to find transpose of matrix

Code:


public class MatrixTranspose {
    public static void main(String[] args) {
        int[][] original = { {1, 2, 3}, {4, 5, 6} };
        int rows = 2;
        int cols = 3;

        int[][] transpose = new int[cols][rows];

        for (int i = 0; i < rows; i++) {
            for (int j = 0; j < cols; j++) {
                transpose[j][i] = original[i][j];
            }
        }

        System.out.println("Transposed Matrix:");
        for (int i = 0; i < cols; i++) {
            for (int j = 0; j < rows; j++) {
                System.out.print(transpose[i][j] + " ");
            }
            System.out.println();
        }
    }
}

abstract class NumberOperation {
    protected double number;
    public NumberOperation(double num) { this.number = num; }
    abstract void calculate();
}

class Square extends NumberOperation {
    public Square(double num) { super(num); }
    @Override
    void calculate() {
        System.out.println("Square of " + number + " is: " + (number * number));
    }
}

class SquareRoot extends NumberOperation {
    public SquareRoot(double num) { super(num); }
    @Override
    void calculate() {
        System.out.println("Square root of " + number + " is: " + Math.sqrt(number));
    }
}

class Cube extends NumberOperation {
    public Cube(double num) { super(num); }
    @Override
    void calculate() {
        System.out.println("Cube of " + number + " is: " + (number * number * number));
    }
}

public class AbstractDemo {
    public static void main(String[] args) {
        NumberOperation op1 = new Square(5);
        NumberOperation op2 = new SquareRoot(25);
        NumberOperation op3 = new Cube(3);

Code:


<!DOCTYPE html>
<html lang="en">
<head>
    <title>To-Do List</title>
    <style>
        /* --- CSS --- */
        body { font-family: Arial, sans-serif; background: #f4f4f4; }
        .container {
            width: 400px;
            margin: 20px auto;
            background: #fff;
            padding: 20px;
            border-radius: 8px;
        }
        li {
            padding: 10px;
            border-bottom: 1px solid #ddd;
            cursor: pointer;
        }
        li:hover { background: #f0f0f0; }
    </style>
</head>
<body>

    <!-- --- HTML --- -->
    <div class="container">
        <h2>My To-Do List</h2>
        <input type="text" id="taskInput" placeholder="New task...">
        <button onclick="addTask()">Add</button>
        <ul id="taskList"></ul>
    </div>

    <!-- --- JavaScript --- -->
    <script>
        function addTask() {
            let inputField = document.getElementById("taskInput");
            let taskText = inputField.value.trim();

            if (taskText === "") return;

            let li = document.createElement("li");
            li.textContent = taskText + " (click to delete)";

            // (EASY DELETE) Click the task to remove it
            li.onclick = function() {
                this.remove();
            };

            document.getElementById("taskList").appendChild(li);
            inputField.value = "";
        }
    </script>

</body>
</html>

27. Sum of 3 and 5 Multiples (Combined)

Problem Statement:

Write a JavaScript program to display sum of 3 and 5 multiples under 1000(use function).

Code:


<!DOCTYPE html>
<html lang="en">
<body>
    <h2>Sum of 3 and 5 multiples under 1000</h2>
    <p id="result">Calculating...</p>
    
    <script>
        function sumMultiples() {
            let sum = 0;
            for (let i = 1; i < 1000; i++) {
                if (i % 3 === 0 || i % 5 === 0) {
                    sum += i;
                }
            }
            return sum;
        }
        let total = sumMultiples();
        document.getElementById("result").textContent = "Result: " + total;
    </script>
</body>
</html>

28. Product of Digits (Combined)

Problem Statement:

Write a JavaScript program to display product of digits of a number(use function).

Code:


<!DOCTYPE html>
<html lang="en">
<body>
    <h2>Product of digits for 12345</h2>
    <p id="result">Calculating...</p>
    
    <script>
        function getProductOfDigits(number) {
            let product = 1;
            let numString = String(number);
            for (let i = 0; i < numString.length; i++) {
                let digit = parseInt(numString[i]);
                product *= digit;
            }
            return product;
        }
        let inputNum = 12345;
        let product = getProductOfDigits(inputNum);
        document.getElementById("result").textContent = `Product for ${inputNum} is: ${product}`;
    </script>
</body>
</html>

29. Form Validation (Combined)

Problem Statement:

Write a JavaScript use function Program to validate Form

Code:


<!DOCTYPE html>
<html lang="en">
<head>
    <title>Form Validation</title>
    <style>
        /* --- CSS --- */
        body { font-family: Arial, sans-serif; padding: 20px; }
        form { padding: 20px; background: #f4f4f4; }
        div { margin-bottom: 15px; }
        .error { color: red; font-size: 0.9em; }
    </style>
</head>
<body>

    <!-- --- HTML --- -->
    <form id="myForm" onsubmit="return validateForm()">
        <h2>Login Form</h2>
        <div>
            Username: <input type="text" id="username">
        </div>
        <div>
            Password (min 6): <input type="password" id="password">
        </div>
        <p id="error-message" class="error"></p>
        <button type="submit">Submit</button>
    </form>

    <!-- --- JavaScript --- -->
    <script>
        function validateForm() {
            let username = document.getElementById("username").value.trim();
            let password = document.getElementById("password").value.trim();
            let errorMsg = document.getElementById("error-message");

            if (username === "") {
                errorMsg.textContent = "Username is required!";
                return false; // Stop submission
            }
            if (password.length < 6) {
                errorMsg.textContent = "Password must be at least 6 characters.";
                return false; // Stop submission
            }

            errorMsg.textContent = "";
            alert("Form is valid! Submitting...");
            return true; // Allow submission
        }
    </script>

</body>
</html>
